def epsilon_greedy_uncertainty_sampling(gp, X_candidate, epsilon=0.1):
    """
    ŒµgreedyÊà¶Áï•„ÇíÁµÑ„ÅøËæº„Çì„Å†‰∏çÁ¢∫ÂÆüÊÄß„Çµ„É≥„Éó„É™„É≥„Ç∞

    Parameters:
    -----------
    gp : GaussianProcessRegressor
        Â≠¶ÁøíÊ∏à„ÅøGP„É¢„Éá„É´
    X_candidate : array (n_candidates, n_features)
        ÂÄôË£úÁÇπ
    epsilon : float
        „É©„É≥„ÉÄ„É†Êé¢Á¥¢„ÅÆÁ¢∫ÁéáÔºà0~1Ôºâ

    Returns:
    --------
    next_x : array
        Ê¨°„ÅÆ„Çµ„É≥„Éó„É™„É≥„Ç∞ÁÇπ
    """
    if np.random.rand() < epsilon:
        # ŒµÁ¢∫Áéá„Åß„É©„É≥„ÉÄ„É†„Çµ„É≥„Éó„É™„É≥„Ç∞

        print(f"  „É©„É≥„ÉÄ„É†Êé¢Á¥¢ÔºàŒµ={epsilon}Ôºâ")
    else:
        # (1-Œµ)Á¢∫Áéá„Åß‰∏çÁ¢∫ÂÆüÊÄß„Çµ„É≥„Éó„É™„É≥„Ç∞

    next_x = X_candidate[next_idx]
    return next_x, next_idx

from sklearn.cluster import KMeans

def fast_diversity_sampling(X_sampled, X_candidate, n_clusters=10):
    """
    k-means „ÇØ„É©„Çπ„Çø„É™„É≥„Ç∞„Å´„Çà„ÇãÈ´òÈÄüÂ§öÊßòÊÄß„Çµ„É≥„Éó„É™„É≥„Ç∞

    Parameters:
    -----------
    X_sampled : array (n_sampled, n_features)
        Êó¢Â≠ò„Çµ„É≥„Éó„É´
    X_candidate : array (n_candidates, n_features)
        ÂÄôË£úÁÇπ
    n_clusters : int
        „ÇØ„É©„Çπ„ÇøÊï∞

    Returns:
    --------
    next_x : array
        Ê¨°„ÅÆ„Çµ„É≥„Éó„É™„É≥„Ç∞ÁÇπ
    """
    # ÂÄôË£úÁÇπ„Çí„ÇØ„É©„Çπ„Çø„É™„É≥„Ç∞
    kmeans = KMeans(n_clusters=n_clusters, random_state=42)
    kmeans.fit(X_candidate)

    # ÂêÑ„ÇØ„É©„Çπ„Çø‰∏≠ÂøÉ„Åã„ÇâÊúÄ„ÇÇÈÅ†„ÅÑÂÄôË£úÁÇπ„ÇíÈÅ∏Êäû
    cluster_centers = kmeans.cluster_centers_
    distances_from_sampled = np.min(
        np.linalg.norm(
            cluster_centers[:, np.newaxis, :] -
            X_sampled[np.newaxis, :, :],
            axis=2
        ),
        axis=1
    )

    # ÊúÄ„ÇÇÈÅ†„ÅÑ„ÇØ„É©„Çπ„Çø„ÅÆ‰ª£Ë°®ÁÇπ„ÇíÈÅ∏Êäû
    farthest_cluster = np.argmax(distances_from_sampled)
    cluster_mask = (kmeans.labels_ == farthest_cluster)
    candidates_in_cluster = X_candidate[cluster_mask]

    # „ÇØ„É©„Çπ„ÇøÂÜÖ„Åß„ÇØ„É©„Çπ„Çø‰∏≠ÂøÉ„Å´ÊúÄ„ÇÇËøë„ÅÑÁÇπ„ÇíÈÅ∏Êäû
    distances_to_center = np.linalg.norm(
        candidates_in_cluster - cluster_centers[farthest_cluster],
        axis=1
    )
    next_idx_in_cluster = np.argmin(distances_to_center)
    next_x = candidates_in_cluster[next_idx_in_cluster]

    return next_x

class RobustClosedLoopOptimizer:
    """
    ÂÆüÈ®ìÂ§±Êïó„Å´ÂØæÂøú„Åó„Åü„ÇØ„É≠„Éº„Ç∫„Éâ„É´„Éº„ÉóÊúÄÈÅ©Âåñ
    """

    def __init__(self, objective_function, total_budget=50, failure_rate=0.1):
        """
        Parameters:
        -----------
        objective_function : callable
            ÁõÆÁöÑÈñ¢Êï∞ÔºàÂÆüÈ®ì„Ç∑„Éü„É•„É¨„Éº„Çø„ÉºÔºâ
        total_budget : int
            Á∑èÂÆüÈ®ì‰∫àÁÆó
        failure_rate : float
            ÂÆüÈ®ìÂ§±ÊïóÁéáÔºà0~1Ôºâ
        """
        self.objective_function = objective_function
        self.total_budget = total_budget
        self.failure_rate = failure_rate

        self.X_sampled = []
        self.y_observed = []
        self.failures = []

    def execute_experiment(self, x):
        """
        ÂÆüÈ®ìÂÆüË°åÔºàÂ§±Êïó„ÅÆÂèØËÉΩÊÄß„ÅÇ„ÇäÔºâ

        Returns:
        --------
        success : bool
            ÂÆüÈ®ìÊàêÂäü„Éï„É©„Ç∞
        result : float or None
            ÊàêÂäüÊôÇ„ÅØÊ∏¨ÂÆöÂÄ§„ÄÅÂ§±ÊïóÊôÇ„ÅØNone
        """
        # Â§±Êïó„Ç∑„Éü„É•„É¨„Éº„Ç∑„Éß„É≥
        if np.random.rand() < self.failure_rate:
            print(f"  ÂÆüÈ®ìÂ§±Êïó: x={x}")
            return False, None

        # ÊàêÂäüÊôÇ„ÅØÁõÆÁöÑÈñ¢Êï∞Ë©ï‰æ°
        y = self.objective_function(x)
        return True, y

    def run(self):
        """„ÇØ„É≠„Éº„Ç∫„Éâ„É´„Éº„ÉóÊúÄÈÅ©ÂåñÂÆüË°å"""
        # ÂàùÊúüÂåñ
        X_init = np.random.uniform(0, 1, (5, 1))
        for x in X_init:
            success, y = self.execute_experiment(x)
            if success:
                self.X_sampled.append(x)
                self.y_observed.append(y)
                self.failures.append(False)
            else:
                self.failures.append(True)

        # „É°„Ç§„É≥„É´„Éº„Éó
        experiments_done = len(X_init)

        while len(self.y_observed) < self.total_budget:
            if experiments_done >= self.total_budget * 1.5:
                print("ÂÆüÈ®ì‰∫àÁÆóË∂ÖÈÅéÔºàÂ§±ÊïóÂ§öÊï∞Ôºâ")
                break

            # GP„É¢„Éá„É´Â≠¶Áøí
            if len(self.y_observed) < 3:
                # „Éá„Éº„Çø‰∏çË∂≥ÊôÇ„ÅØ„É©„É≥„ÉÄ„É†„Çµ„É≥„Éó„É™„É≥„Ç∞
                next_x = np.random.uniform(0, 1, (1, 1))
                print(f"„Éá„Éº„Çø‰∏çË∂≥: „É©„É≥„ÉÄ„É†„Çµ„É≥„Éó„É™„É≥„Ç∞")
            else:
                kernel = ConstantKernel(1.0) * RBF(length_scale=0.15)
                gp = GaussianProcessRegressor(kernel=kernel)
                X_array = np.array(self.X_sampled)
                y_array = np.array(self.y_observed)
                gp.fit(X_array, y_array)

                # EIÊúÄÂ§ßÂåñ
                X_candidate = np.linspace(0, 1, 500).reshape(-1, 1)
                mu, sigma = gp.predict(X_candidate, return_std=True)
                f_best = np.max(y_array)

                from scipy.stats import norm
                improvement = mu - f_best - 0.01
                Z = improvement / (sigma + 1e-9)
                ei = improvement * norm.cdf(Z) + sigma * norm.pdf(Z)

                next_idx = np.argmax(ei)
                next_x = X_candidate[next_idx:next_idx+1]

            # ÂÆüÈ®ìÂÆüË°å
            success, y = self.execute_experiment(next_x)
            experiments_done += 1

            if success:
                self.X_sampled.append(next_x)
                self.y_observed.append(y)
                self.failures.append(False)
                print(f"ÊàêÂäü {len(self.y_observed)}/{self.total_budget}: "
                      f"x={next_x[0][0]:.3f}, y={y:.3f}")
            else:
                self.failures.append(True)
                print(f"Â§±Êïó: ÂÜçË©¶Ë°å„Åó„Åæ„Åô")

        # ÁµêÊûú„Çµ„Éû„É™„Éº
        success_rate = len(self.y_observed) / experiments_done
        print(f"\nÊúÄÁµÇÁµêÊûú:")
        print(f"  Á∑èÂÆüÈ®ìÂõûÊï∞: {experiments_done}")
        print(f"  ÊàêÂäüÂÆüÈ®ìÊï∞: {len(self.y_observed)}")
        print(f"  ÊàêÂäüÁéá: {success_rate:.1%}")
        print(f"  ÊúÄËâØÂÄ§: {np.max(self.y_observed):.4f}")
