class Perceptron:
    """æ±ç”¨ãƒ‘ãƒ¼ã‚»ãƒ—ãƒˆãƒ­ãƒ³ã‚¯ãƒ©ã‚¹"""

            bias: ãƒã‚¤ã‚¢ã‚¹å€¤
        """
        self.w = np.array(weights)

    def forward(self, x):
        """
        é †ä¼æ’­ï¼ˆforward propagationï¼‰

        Args:
            x: å…¥åŠ›å€¤ã®é…åˆ—

        Returns:
            0 ã¾ãŸã¯ 1

    def __call__(self, *inputs):
        """å‘¼ã³å‡ºã—å¯èƒ½ã«ã™ã‚‹"""
        x = np.array(inputs)
        return self.forward(x)
