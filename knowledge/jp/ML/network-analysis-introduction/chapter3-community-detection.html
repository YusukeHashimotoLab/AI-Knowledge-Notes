def label_propagation_manual(G, max_iter=100, seed=None):
    """Label Propagationã®æ‰‹å‹•å®Ÿè£…ï¼ˆæ•™è‚²ç›®çš„ï¼‰"""
    if seed is not None:
        np.random.seed(seed)

    # åˆæœŸåŒ–: å„ãƒãƒ¼ãƒ‰ã«ä¸€æ„ã®ãƒ©ãƒ™ãƒ«
    labels = {node: i for i, node in enumerate(G.nodes())}
    nodes = list(G.nodes())

    for iteration in range(max_iter):
        # ãƒãƒ¼ãƒ‰ã®å‡¦ç†é †åºã‚’ãƒ©ãƒ³ãƒ€ãƒ åŒ–
        np.random.shuffle(nodes)
        changed = False

        for node in nodes:
            # éš£æ¥ãƒãƒ¼ãƒ‰ã®ãƒ©ãƒ™ãƒ«ã‚’åé›†
            neighbor_labels = [labels[neighbor] for neighbor in G.neighbors(node)]

            label_counts = Counter(neighbor_labels)
            max_count = max(label_counts.values())
            most_common = [label for label, count in label_counts.items()
                          if count == max_count]
            new_label = np.random.choice(most_common)

            if labels[node] != new_label:
                labels[node] = new_label
                changed = True

        if not changed:
            print(f"åæŸ: {iteration + 1}å›ã®åå¾©")
            break
