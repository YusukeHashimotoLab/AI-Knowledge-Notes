    # Stage III: æˆé•·ã«ã‚ˆã‚‹ãƒ¢ãƒãƒãƒ¼æ¶ˆè²»
    if N > 0 and r_avg > 0 and C > C_s:
        growth_rate = (D * v_m * (C - C_s)) / r_avg
        dC_growth = -N * 4 * np.pi * r_avg**2 * growth_rate / v_m
        dr_dt = growth_rate
    else:
        dC_growth = 0
        dr_dt = 0

    dC_dt = dC_precursor + dC_nucleation + dC_growth
    dN_dt = J_nucleation

    return [dC_dt, dN_dt, dr_dt]

# åˆæœŸæ¡ä»¶ [C, N, r_avg]
y0 = [0, 0, 2e-9]  # åˆæœŸãƒ¢ãƒãƒãƒ¼æ¿ƒåº¦0, æ ¸æ•°0, åˆæœŸåŠå¾„2 nm

# æ™‚é–“ç¯„å›²
t = np.linspace(0, 100, 1000)  # 0ï½100ç§’

# å¾®åˆ†æ–¹ç¨‹å¼ã‚’è§£ã
