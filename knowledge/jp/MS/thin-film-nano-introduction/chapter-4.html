def cahn_hilliard_2d(Nx, Ny, dt, n_steps, M=1.0, kappa=0.5):
    """
    2æ¬¡å…ƒCahn-Hilliardæ–¹ç¨‹å¼ã«ã‚ˆã‚‹ç›¸åˆ†é›¢ã‚·ãƒŸãƒ¥ãƒ¬ãƒ¼ã‚·ãƒ§ãƒ³

    """

    np.random.seed(42)

        laplacian_c = (np.roll(c, 1, axis=0) + np.roll(c, -1, axis=0) +

        if step % 100 == 0:

            print(f"Step {step}/{n_steps}: å¹³å‡æ¿ƒåº¦ {c.mean():.3f}")

    return c_history
