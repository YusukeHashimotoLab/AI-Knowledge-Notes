def weibull_cumulative_probability(sigma, m, sigma_0, sigma_u=0):
    """
    Weibullç´¯ç©ç ´å£Šç¢ºç‡ã®è¨ˆç®—

    Parameters:
    -----------
    sigma : float or array

    m : float

def estimate_weibull_parameters(strength_data):
    """
    å®Ÿæ¸¬å¼·åº¦ãƒ‡ãƒ¼ã‚¿ã‹ã‚‰Weibullãƒ‘ãƒ©ãƒ¡ãƒ¼ã‚¿ã‚’æ¨å®š

    Parameters:
    -----------
    strength_data : array
        ç ´å£Šå¼·åº¦ãƒ‡ãƒ¼ã‚¿ [Pa]

    Returns:
    --------
    m : float
        Weibullãƒ¢ã‚¸ãƒ¥ãƒ©ã‚¹
    sigma_0 : float
        ç‰¹æ€§å¼·åº¦ [Pa]
    R_squared : float
        æ±ºå®šä¿‚æ•°ï¼ˆãƒ•ã‚£ãƒƒãƒˆç²¾åº¦ï¼‰
    """
    # ãƒ‡ãƒ¼ã‚¿ã®ã‚½ãƒ¼ãƒˆ
    sorted_strength = np.sort(strength_data)
    n = len(sorted_strength)

    # ç ´å£Šç¢ºç‡ã®æ¨å®šï¼ˆä¸­å¤®å€¤ãƒ©ãƒ³ã‚¯æ³•ï¼‰
    P_f = np.array([(i - 0.3) / (n + 0.4) for i in range(1, n + 1)])

    # Weibullå¤‰æ›: Y = ln ln(1/(1-P_f)), X = ln(Ïƒ)
    # 1ã«æ¥µã‚ã¦è¿‘ã„å€¤ã‚„0ã«æ¥µã‚ã¦è¿‘ã„å€¤ã‚’é¿ã‘ã‚‹
    valid_indices = (P_f > 0.001) & (P_f < 0.999)
    P_f_valid = P_f[valid_indices]
    sigma_valid = sorted_strength[valid_indices]

    Y = np.log(-np.log(1 - P_f_valid))
    X = np.log(sigma_valid)

    # ç·šå½¢å›å¸°
    coeffs = np.polyfit(X, Y, 1)
    m = coeffs[0]
    sigma_0 = np.exp(-coeffs[1] / m)

    # æ±ºå®šä¿‚æ•°RÂ²ã®è¨ˆç®—
    Y_pred = m * X + coeffs[1]
    SS_res = np.sum((Y - Y_pred)**2)
    SS_tot = np.sum((Y - np.mean(Y))**2)
    R_squared = 1 - (SS_res / SS_tot)

    return m, sigma_0, R_squared
