def calculate_heating_energy(material, T_initial, T_final, mass_kg):
    """

        å„æ®µéšã®ã‚¨ãƒãƒ«ã‚®ãƒ¼å†…è¨³
    """
    data = materials_database[material]
    n_moles = (mass_kg * 1000) / data['atomic_mass']  # ç‰©è³ªé‡ï¼ˆmolï¼‰

    energy_breakdown = {
        'solid_heating': 0,
        'melting': 0,
        'liquid_heating': 0,
        'vaporization': 0,
        'gas_heating': 0,
        'total': 0
    }

    T = T_initial

    # 1. å›ºä½“åŠ ç†±ï¼ˆT_initial â†’ T_meltï¼‰
    if T < data['T_melt'] and T_final > T:

        T = T_end_solid

    # 2. èè§£ï¼ˆT_meltï¼‰
    if T >= data['T_melt'] and T_final > T:

        # T ã¯å¤‰ã‚ã‚‰ãªã„ï¼ˆç›¸è»¢ç§»ï¼‰

    # 3. æ¶²ä½“åŠ ç†±ï¼ˆT_melt â†’ T_boilï¼‰

    energy_breakdown['total'] = sum(energy_breakdown.values())

    return energy_breakdown, n_moles
