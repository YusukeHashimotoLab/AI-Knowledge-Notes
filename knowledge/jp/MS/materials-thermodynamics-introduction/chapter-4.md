#### ğŸ’¡ ãƒ¬ãƒãƒ¼ãƒ«ãƒ¼ãƒ«ï¼ˆLever Ruleï¼‰ã®åŸç†

#### ã‚³ãƒ¼ãƒ‰ä¾‹2: Al-Siå…±æ™¶ç³»ç›¸å›³ã¨ãƒ¬ãƒãƒ¼ãƒ«ãƒ¼ãƒ«è¨ˆç®—

    # Al-Siç³»ã®ç›¸å›³ãƒ‡ãƒ¼ã‚¿ï¼ˆç°¡ç•¥åŒ–ï¼‰
    def create_al_si_phase_diagram():

        # çµ„æˆç¯„å›²
        comp = np.linspace(0, 100, 300)
    
        # æ¶²ç›¸ç·šï¼ˆAlå´ã¨Siå´ï¼‰

        Si_side = 1414 - 9.0 * (100 - comp[150:])  # Siå´ã®æ¶²ç›¸ç·š
        liquidus = np.concatenate([Al_side, Si_side])
    
        # å…±æ™¶ç‚¹
        eutectic_comp = 12.6  # wt% Si
        eutectic_temp = 577   # â„ƒ
    
        # å›ºæº¶åº¦é™ï¼ˆæ¸©åº¦ä¾å­˜æ€§ã‚’ç°¡ç•¥åŒ–ï¼‰
        solubility_Al = np.full(300, 1.65)  # Î±ç›¸ä¸­ã®Siå›ºæº¶åº¦é™
        solubility_Si = np.full(300, 0.05)  # Siç›¸ä¸­ã®Alå›ºæº¶åº¦é™ï¼ˆã»ã¼0ï¼‰

                        alpha=0.3, color='yellow', label='L + Î±')

                        alpha=0.3, color='orange', label='L + Î²(Si)')

                'g--', linewidth=2, label='å…±æ™¶ç·š')
    
        # å…±æ™¶ç‚¹

                    xytext=(eutectic_comp + 15, eutectic_temp + 80),
                    arrowprops=dict(arrowstyle='->', color='black', lw=2),
                    fontsize=11, ha='center', fontweight='bold',
                    bbox=dict(boxstyle='round,pad=0.5', facecolor='yellow', alpha=0.7))

        ax.plot([100-solubility_Si[0], 100-solubility_Si[0]], [0, eutectic_temp],
                'b--', linewidth=1.5, alpha=0.7)
    
        # ãƒ¬ãƒãƒ¼ãƒ«ãƒ¼ãƒ«é©ç”¨ä¾‹ï¼ˆ8 wt% Si, 600â„ƒï¼‰

        ax.plot([solubility_Al[0], liquidus_comp_at_temp],
                [example_temp, example_temp], 'ko', markersize=6)

        print(f"æ¶²ç›¸åˆ†ç‡: {f_L:.3f} ({f_L*100:.1f}%)")
        print(f"Î±ç›¸åˆ†ç‡: {f_alpha:.3f} ({f_alpha*100:.1f}%)")

### 2.2 Pb-Snå…±æ™¶ç³»ã¨å†·å´æ›²ç·š

    import numpy as np
    import matplotlib.pyplot as plt
    
    def simulate_cooling_curve(composition_Sn, initial_temp=300, final_temp=100,
                               cooling_rate=0.5, latent_heat_factor=20):
        """
        Pb-Snç³»ã®å†·å´æ›²ç·šã‚’ã‚·ãƒŸãƒ¥ãƒ¬ãƒ¼ãƒˆ
    
        Parameters:
        -----------
        composition_Sn : float
            Snçµ„æˆï¼ˆwt%ï¼‰
        initial_temp : float
            åˆæœŸæ¸©åº¦ï¼ˆâ„ƒï¼‰
        final_temp : float
            æœ€çµ‚æ¸©åº¦ï¼ˆâ„ƒï¼‰
        cooling_rate : float
            å†·å´é€Ÿåº¦ï¼ˆâ„ƒ/sï¼‰
        latent_heat_factor : float
            æ½œç†±ã«ã‚ˆã‚‹æ¸©åº¦åœç•™ã®å¼·ã•
        """
        # ç›¸å›³ãƒ‘ãƒ©ãƒ¡ãƒ¼ã‚¿
        eutectic_comp = 61.9  # wt% Sn
        eutectic_temp = 183   # â„ƒ
        Pb_melting = 327      # â„ƒ
        Sn_melting = 232      # â„ƒ
    
        # æ¶²ç›¸ç·šæ¸©åº¦ã‚’çµ„æˆã‹ã‚‰æ¨å®š
        if composition_Sn < eutectic_comp:
            liquidus_temp = Pb_melting - (Pb_melting - eutectic_temp) * (composition_Sn / eutectic_comp)
        else:
            liquidus_temp = Sn_melting - (Sn_melting - eutectic_temp) * ((100 - composition_Sn) / (100 - eutectic_comp))
    
        # æ™‚é–“é…åˆ—
        time = np.arange(0, (initial_temp - final_temp) / cooling_rate, 0.1)
        temperature = np.zeros_like(time)
    
        for i, t in enumerate(time):
            # åŸºæœ¬å†·å´
            temp = initial_temp - cooling_rate * t
    
            # æ¶²ç›¸ç·šã§ã®åœç•™ï¼ˆå‡å›ºé–‹å§‹ï¼‰
            if abs(temp - liquidus_temp) < 5:
                temp += latent_heat_factor * np.exp(-((temp - liquidus_temp)**2) / 10)
    
            # å…±æ™¶æ¸©åº¦ã§ã®åœç•™
            if abs(temp - eutectic_temp) < 5:
                # å…±æ™¶çµ„æˆã«è¿‘ã„ã»ã©åœç•™ãŒé¡•è‘—
                eutectic_factor = 1.0 - abs(composition_Sn - eutectic_comp) / eutectic_comp
                temp += latent_heat_factor * eutectic_factor * np.exp(-((temp - eutectic_temp)**2) / 10)
    
            temperature[i] = temp
    
        return time, temperature
    
    # ç•°ãªã‚‹çµ„æˆã§ã®å†·å´æ›²ç·šã‚’è¨ˆç®—

    colors = ['blue', 'green', 'red', 'purple']
    labels = [f'{comp} wt% Sn' for comp in compositions]
    
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(14, 6))
    
    # å†·å´æ›²ç·šã®ãƒ—ãƒ­ãƒƒãƒˆ
    for comp, color, label in zip(compositions, colors, labels):
        time, temp = simulate_cooling_curve(comp)
        if comp == 61.9:
            ax1.plot(time, temp, color=color, linewidth=2.5, label=label + ' (å…±æ™¶)', linestyle='-')
        else:
            ax1.plot(time, temp, color=color, linewidth=2, label=label, alpha=0.8)
    
    ax1.axhline(y=183, color='red', linestyle='--', linewidth=1.5, alpha=0.5, label='å…±æ™¶æ¸©åº¦ (183â„ƒ)')
    ax1.set_xlabel('æ™‚é–“ (s)', fontsize=12, fontweight='bold')
    ax1.set_ylabel('æ¸©åº¦ (â„ƒ)', fontsize=12, fontweight='bold')
    ax1.set_title('Pb-Snç³»ã®å†·å´æ›²ç·š', fontsize=14, fontweight='bold')
    ax1.legend(loc='upper right', fontsize=10)
    ax1.grid(True, alpha=0.3)
    ax1.set_ylim(100, 350)
    
    # å†·å´é€Ÿåº¦ï¼ˆæ¸©åº¦ã®æ™‚é–“å¾®åˆ†ï¼‰
    ax2.set_title('å†·å´é€Ÿåº¦ (dT/dt)', fontsize=14, fontweight='bold')
    for comp, color, label in zip(compositions, colors, labels):

        cooling_rate = np.gradient(temp, time)
        if comp == 61.9:
            ax2.plot(temp, -cooling_rate, color=color, linewidth=2.5, label=label + ' (å…±æ™¶)')
        else:
            ax2.plot(temp, -cooling_rate, color=color, linewidth=2, label=label, alpha=0.8)
    
    ax2.axvline(x=183, color='red', linestyle='--', linewidth=1.5, alpha=0.5)
    ax2.set_xlabel('æ¸©åº¦ (â„ƒ)', fontsize=12, fontweight='bold')
    ax2.set_ylabel('å†·å´é€Ÿåº¦ã®çµ¶å¯¾å€¤ (â„ƒ/s)', fontsize=12, fontweight='bold')
    ax2.legend(loc='upper right', fontsize=10)
    ax2.grid(True, alpha=0.3)
    ax2.set_xlim(100, 350)
    
    plt.tight_layout()
    plt.savefig('pb_sn_cooling_curves.png', dpi=300, bbox_inches='tight')
    plt.show()

    print("1. æ¶²ç›¸ç·šã§ã®åœç•™: å‡å›ºé–‹å§‹ã«ã‚ˆã‚Šæ½œç†±æ”¾å‡º")
