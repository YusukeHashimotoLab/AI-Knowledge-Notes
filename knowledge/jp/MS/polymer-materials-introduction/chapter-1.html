def flory_schulz_distribution(n, p):
    """
    Flory-Schulzåˆ†å¸ƒï¼ˆæœ€ç¢ºåˆ†å¸ƒï¼‰ã‚’è¨ˆç®—
    
    Parameters:
    -----------
    n : int or array
        é‡åˆåº¦
    p : float
        å®˜èƒ½åŸºè»¢åŒ–ç‡
    
    Returns:
    --------
    N_n : float or array
        ãƒ¢ãƒ«åˆ†ç‡
    w_n : float or array

    """
    N_n = n * (p ** (n - 1)) * ((1 - p) ** 2)
    w_n = n**2 * (p ** (n - 1)) * ((1 - p) ** 3)
    return N_n, w_n

    """
    Flory-Schulzåˆ†å¸ƒã®å¹³å‡å€¤ã‚’è¨ˆç®—

    """
    DPn = 1 / (1 - p)
    DPw = (1 + p) / (1 - p)
