def sintering_profile_simulation(heating_rate, sintering_temp, holding_time, cooling_rate):
    """

    holding_time_min = holding_time * 60
    cooling_time = sintering_temp / cooling_rate

    t_heating = np.linspace(0, heating_time, 100)
    T_heating = heating_rate * t_heating

    # ä¿æŒæ®µéš
    t_holding = np.linspace(heating_time, heating_time + holding_time_min, 200)
    T_holding = np.full_like(t_holding, sintering_temp)

    # å†·å´æ®µéš
    t_cooling = np.linspace(heating_time + holding_time_min,
                             heating_time + holding_time_min + cooling_time, 100)
    T_cooling = sintering_temp - cooling_rate * (t_cooling - (heating_time + holding_time_min))

    # å…¨ä½“ã®ãƒ—ãƒ­ãƒ•ã‚¡ã‚¤ãƒ«
    time = np.concatenate([t_heating, t_holding, t_cooling])
    temperature = np.concatenate([T_heating, T_holding, T_cooling])

    # ç›¸å¯¾å¯†åº¦ã®å¤‰åŒ–ï¼ˆçµŒé¨“çš„ãƒ¢ãƒ‡ãƒ«ï¼‰
    density = np.zeros_like(time)

        density[len(t_heating) + i] = 55 + 40 * (1 - np.exp(-5 * tau))

    # å†·å´æ®µéšã§ã¯å¯†åº¦å¤‰åŒ–ãªã—
    density[len(t_heating) + len(t_holding):] = density[len(t_heating) + len(t_holding) - 1]

    # ç²’å¾„ã®å¤‰åŒ–ï¼ˆç²’æˆé•·ï¼‰
    grain_size = np.ones_like(time) * 1.0  # åˆæœŸç²’å¾„1 Î¼m

    # ä¿æŒæ®µéšã§ç²’æˆé•·ï¼ˆG^3 - G0^3 = k*tï¼‰

    # å†·å´æ®µéšã§ã¯ç²’å¾„å¤‰åŒ–ãªã—
    grain_size[len(t_heating) + len(t_holding):] = grain_size[len(t_heating) + len(t_holding) - 1]

    return time, temperature, density, grain_size
