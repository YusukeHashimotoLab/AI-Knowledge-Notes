def temperature_profile(t, T_target, heating_rate, hold_time, cooling_rate):
    """æ¸©åº¦ãƒ—ãƒ­ãƒ•ã‚¡ã‚¤ãƒ«ã‚’ç”Ÿæˆ

def jander_model(t, k):
    """Janderå¼ã«ã‚ˆã‚‹åå¿œç‡è¨ˆç®—

    Args:
        t (array): æ™‚é–“ [hours]
        k (float): é€Ÿåº¦å®šæ•°

    """
    # [1 - (1-Î±)^(1/3)]Â² = kt ã‚’ Î± ã«ã¤ã„ã¦è§£ã
    kt = k * t
    alpha = 1 - (1 - np.sqrt(kt))**3

    return alpha

    kt = k * t
