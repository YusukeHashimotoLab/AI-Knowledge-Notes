import numpy as np
from collections import Counter

def design_targeting_peptide(
    target_receptor,
    peptide_length=10
):
    """
    „Çø„Éº„Ç≤„ÉÜ„Ç£„É≥„Ç∞„Éö„Éó„ÉÅ„Éâ„ÇíË®≠Ë®à

    peptide_length : int

    # Êó¢Áü•„ÅÆ„Çø„Éº„Ç≤„ÉÜ„Ç£„É≥„Ç∞ÈÖçÂàó„É¢„ÉÅ„Éº„Éï
    motifs = {
        'RGD': 'RGD',  # „Ç§„É≥„ÉÜ„Ç∞„É™„É≥ÁµêÂêà
        'folate': 'KKKK',  # ËëâÈÖ∏ÂèóÂÆπ‰ΩìÔºàÊ≠£ÈõªËç∑Ôºâ
        'transferrin': 'HAIYPRH'  # „Éà„É©„É≥„Çπ„Éï„Çß„É™„É≥ÂèóÂÆπ‰Ωì
    }

    if target_receptor in motifs:

        # ÊÆã„Çä„ÅÆÊÆãÂü∫„Åß„Éë„Éá„Ç£„É≥„Ç∞
        remaining = peptide_length - len(core_motif)

            linker = 'GS' * (remaining // 2)

                      linker[remaining//2:remaining]

            f"Êú™Áü•„ÅÆÊ®ôÁöÑ: {target_receptor}"
