import numpy as np
from Bio.PDB import PDBParser
import matplotlib.pyplot as plt

def extract_plddt_scores(pdb_file):
    """AlphaFoldÊßãÈÄ†„Åã„ÇâpLDDT„Çπ„Ç≥„Ç¢„ÇíÊäΩÂá∫

    pLDDTÔºàpredicted Local Distance Difference TestÔºâ„ÅØ
    ÂêÑÊÆãÂü∫„ÅÆ‰∫àÊ∏¨‰ø°È†ºÂ∫¶„Çí0-100„ÅßÁ§∫„ÅôÊåáÊ®ô„ÄÇ

    Ëß£Èáà:
    - pLDDT > 90: Very high (ÂÆüÈ®ìÁöÑÊßãÈÄ†„Å®ÂêåÁ≠â)
    - pLDDT 70-90: Confident („É¢„Éá„É™„É≥„Ç∞ÂèØËÉΩ)
    - pLDDT 50-70: Low (ÊüîËªü„Å™È†òÂüü„ÅÆÂèØËÉΩÊÄß)
    - pLDDT < 50: Very low (‰ø°È†ºÊÄß„Å™„Åó)

    Args:
        pdb_file (str): AlphaFold PDB„Éï„Ç°„Ç§„É´„Éë„Çπ

    Returns:
        tuple: (ÊÆãÂü∫Áï™Âè∑„É™„Çπ„Éà, pLDDT„Çπ„Ç≥„Ç¢„É™„Çπ„Éà)
    """
    parser = PDBParser(QUIET=True)
    structure = parser.get_structure('protein', pdb_file)

    residue_numbers = []
    plddt_scores = []

    for model in structure:
        for chain in model:
            for residue in chain:
                # B-factorÂàó„Å´pLDDT„ÅåÊ†ºÁ¥ç„Åï„Çå„Å¶„ÅÑ„Çã
                for atom in residue:
                    if atom.name == 'CA':  # CŒ±ÂéüÂ≠ê„ÅÆ„Åø
                        residue_numbers.append(residue.id[1])
                        plddt_scores.append(atom.bfactor)
                        break

    return residue_numbers, plddt_scores

def analyze_plddt(pdb_file):
    """pLDDT„ÅÆÁµ±Ë®àËß£Êûê„Å®ÂèØË¶ñÂåñ"""
    res_nums, plddt = extract_plddt_scores(pdb_file)

    # Áµ±Ë®àÊÉÖÂ†±
    mean_plddt = np.mean(plddt)
    very_high = sum(1 for x in plddt if x > 90)
    confident = sum(1 for x in plddt if 70 <= x <= 90)
    low = sum(1 for x in plddt if x < 70)

    print(f"pLDDTÁµ±Ë®à:")
    print(f"  Âπ≥Âùá„Çπ„Ç≥„Ç¢: {mean_plddt:.2f}")
    print(f"  Very high (>90): {very_high}ÊÆãÂü∫ ({very_high/len(plddt)*100:.1f}%)")
    print(f"  Confident (70-90): {confident}ÊÆãÂü∫ ({confident/len(plddt)*100:.1f}%)")
    print(f"  Low (<70): {low}ÊÆãÂü∫ ({low/len(plddt)*100:.1f}%)")

    # ÂèØË¶ñÂåñ
    plt.figure(figsize=(12, 4))
    plt.plot(res_nums, plddt, linewidth=2)
    plt.axhline(y=90, color='g', linestyle='--', label='Very high threshold')
    plt.axhline(y=70, color='orange', linestyle='--', label='Confident threshold')
    plt.axhline(y=50, color='r', linestyle='--', label='Low threshold')
    plt.xlabel('Residue Number')
    plt.ylabel('pLDDT Score')
    plt.title('AlphaFold Prediction Confidence (pLDDT)')
    plt.legend()
    plt.grid(alpha=0.3)
    plt.ylim(0, 100)
    plt.tight_layout()
    plt.savefig('plddt_analysis.png', dpi=150)
    print("‚úì „Ç∞„É©„Éï‰øùÂ≠ò: plddt_analysis.png")

    return mean_plddt, plddt

    Args:
        sequence (str): „Ç¢„Éü„ÉéÈÖ∏ÈÖçÂàó
        max_hits (int): ÊúÄÂ§ß„Éí„ÉÉ„ÉàÊï∞

    Returns:
        list: Áõ∏ÂêåÈÖçÂàó„ÅÆ„É™„Çπ„Éà
    """
    print("BLAST„Çµ„Éº„ÉÅÈñãÂßãÔºàÊï∞ÂàÜ„Åã„Åã„Çä„Åæ„ÅôÔºâ...")

    # NCBI BLAST„ÅßÊ§úÁ¥¢
    result_handle = NCBIWWW.qblast(
        program="blastp",
        database="nr",
        sequence=sequence,
        hitlist_size=max_hits
    )

    # ÁµêÊûú„Çí„Éë„Éº„Çπ
    blast_records = NCBIXML.parse(result_handle)
    record = next(blast_records)

    homologs = []
    for alignment in record.alignments[:max_hits]:
        for hsp in alignment.hsps:
            if hsp.expect < 1e-5:  # E-valueÈñæÂÄ§

                    'title': alignment.title,
                    'e_value': hsp.expect,
                    'identity': hsp.identities / hsp.align_length,

                })

    print(f"‚úì {len(homologs)}ÂÄã„ÅÆÁõ∏ÂêåÈÖçÂàó„ÇíÊ§úÂá∫")
    return homologs

def calculate_sequence_conservation(msa_sequences):
    """ÈÖçÂàó‰øùÂ≠òÂ∫¶„ÅÆË®àÁÆó

    Args:
        msa_sequences (list): „Ç¢„É©„Ç§„É≥„É°„É≥„ÉàÊ∏à„ÅøÈÖçÂàó„ÅÆ„É™„Çπ„Éà

    Returns:
        np.array: ÂêÑ‰ΩçÁΩÆ„ÅÆ‰øùÂ≠òÂ∫¶„Çπ„Ç≥„Ç¢Ôºà0-1Ôºâ
    """
    if not msa_sequences:
        return None

    length = len(msa_sequences[0])
    conservation = np.zeros(length)

    for pos in range(length):
        # ÂêÑ‰ΩçÁΩÆ„ÅÆ„Ç¢„Éü„ÉéÈÖ∏È†ªÂ∫¶„ÇíË®àÁÆó
        amino_acids = [seq[pos] for seq in msa_sequences if pos < len(seq)]

        # „ÇÆ„É£„ÉÉ„Éó„ÇíÈô§Â§ñ
        amino_acids = [aa for aa in amino_acids if aa != '-']

        if amino_acids:
            # ÊúÄÈ†ª„Ç¢„Éü„ÉéÈÖ∏„ÅÆÂâ≤Âêà = ‰øùÂ≠òÂ∫¶
            from collections import Counter
            most_common = Counter(amino_acids).most_common(1)[0][1]
            conservation[pos] = most_common / len(amino_acids)

    return conservation

from Bio.PDB import PDBParser, NeighborSearch
import numpy as np

def detect_binding_pockets(pdb_file, pocket_threshold=10.0):
    """„Çø„É≥„Éë„ÇØË≥™ÊßãÈÄ†„Åã„ÇâÁµêÂêà„Éù„Ç±„ÉÉ„Éà„ÇíÊ§úÂá∫

    Á∞°ÊòìÂÆüË£Ö: Ë°®Èù¢„ÅÆÂáπ„Çì„Å†È†òÂüü„ÇíÊ§úÂá∫
    ÔºàÂÆüÁî®ÁöÑ„Å´„ÅØ Fpocket, DoGSiteÁ≠â„ÅÆÂ∞ÇÈñÄ„ÉÑ„Éº„É´„ÇíÊé®Â•®Ôºâ

    Args:
        pdb_file (str): PDB„Éï„Ç°„Ç§„É´„Éë„Çπ
        pocket_threshold (float): „Éù„Ç±„ÉÉ„ÉàÂà§ÂÆö„ÅÆË∑ùÈõ¢ÈñæÂÄ§ [√Ö]

    Returns:
        list: „Éù„Ç±„ÉÉ„ÉàÂÄôË£ú„ÅÆÊÆãÂü∫Áï™Âè∑„É™„Çπ„Éà
    """
    parser = PDBParser(QUIET=True)
    structure = parser.get_structure('protein', pdb_file)

    # ÂÖ®ÂéüÂ≠ê„ÇíÂèñÂæó
    atoms = []
    for model in structure:
        for chain in model:
            for residue in chain:
                for atom in residue:
                    atoms.append(atom)

    # ËøëÂÇçÊ§úÁ¥¢„Ç™„Éñ„Ç∏„Çß„ÇØ„Éà„Çí‰ΩúÊàê
    ns = NeighborSearch(atoms)

    # Ë°®Èù¢ÊÆãÂü∫„ÇíÊ§úÂá∫ÔºàÊ∫∂Â™íÈú≤Âá∫Â∫¶„ÅåÈ´ò„ÅÑÔºâ
    surface_residues = []

    for model in structure:
        for chain in model:
            for residue in chain:
                # CŒ±ÂéüÂ≠ê„ÇíÂèñÂæó
                ca_atom = None
                for atom in residue:
                    if atom.name == 'CA':
                        ca_atom = atom
                        break

                if ca_atom is None:
                    continue

                # ÂçäÂæÑ10√Ö‰ª•ÂÜÖ„ÅÆÂéüÂ≠êÊï∞„ÇíÊï∞„Åà„Çã
                neighbors = ns.search(ca_atom.coord, pocket_threshold)

                # ËøëÂÇçÂéüÂ≠ê„ÅåÂ∞ë„Å™„ÅÑ = Ë°®Èù¢„Å´Èú≤Âá∫
                if len(neighbors) < 30:  # ÁµåÈ®ìÁöÑÈñæÂÄ§
                    surface_residues.append({
                        'residue_number': residue.id[1],
                        'residue_name': residue.resname,
                        'chain': chain.id,
                        'neighbors': len(neighbors),
                        'coord': ca_atom.coord
                    })

    # „ÇØ„É©„Çπ„Çø„É™„É≥„Ç∞„Åß„Éù„Ç±„ÉÉ„Éà„Çí„Ç∞„É´„Éº„ÉóÂåñ
    pocket_candidates = cluster_surface_residues(surface_residues)

    print(f"‚úì {len(pocket_candidates)}ÂÄã„ÅÆ„Éù„Ç±„ÉÉ„ÉàÂÄôË£ú„ÇíÊ§úÂá∫")
    for i, pocket in enumerate(pocket_candidates, 1):
        print(f"  Pocket {i}: {len(pocket)}ÊÆãÂü∫")

    return pocket_candidates

def cluster_surface_residues(surface_residues, distance_cutoff=8.0):
    """Ë°®Èù¢ÊÆãÂü∫„Çí„ÇØ„É©„Çπ„Çø„É™„É≥„Ç∞„Åó„Å¶„Éù„Ç±„ÉÉ„Éà„ÇíÂêåÂÆö"""
    if not surface_residues:
        return []

    # Ë∑ùÈõ¢Ë°åÂàó„ÇíË®àÁÆó
    coords = np.array([r['coord'] for r in surface_residues])
    n = len(coords)

    # Á∞°Êòì„ÇØ„É©„Çπ„Çø„É™„É≥„Ç∞ÔºàÂÆüÁî®ÁöÑ„Å´„ÅØDBSCANÁ≠â„Çí‰ΩøÁî®Ôºâ
    visited = set()
    pockets = []

    for i in range(n):
        if i in visited:
            continue

        pocket = [surface_residues[i]]
        visited.add(i)
        queue = [i]

        while queue:
            current = queue.pop(0)
            current_coord = coords[current]

            for j in range(n):
                if j in visited:
                    continue

                distance = np.linalg.norm(current_coord - coords[j])
                if distance < distance_cutoff:
                    pocket.append(surface_residues[j])
                    visited.add(j)
                    queue.append(j)

        if len(pocket) >= 5:  # ÊúÄÂ∞è„Éù„Ç±„ÉÉ„Éà„Çµ„Ç§„Ç∫
            pockets.append(pocket)

    # „Çµ„Ç§„Ç∫È†Ü„Å´„ÇΩ„Éº„Éà
    pockets.sort(key=len, reverse=True)

    return pockets

def calculate_rmsd(pdb1, pdb2):
    """2„Å§„ÅÆ„Çø„É≥„Éë„ÇØË≥™ÊßãÈÄ†„ÅÆRMSD„ÇíË®àÁÆó

    RMSDÔºàRoot Mean Square DeviationÔºâ„ÅØÊßãÈÄ†„ÅÆÈ°û‰ººÂ∫¶„Çí
    Á§∫„ÅôÊåáÊ®ô„ÄÇÂÄ§„ÅåÂ∞è„Åï„ÅÑ„Åª„Å©È°û‰ºº„ÄÇ

    - RMSD 2-5√Ö: È°û‰ººÔºàÂêå„Åò„Éï„Ç©„Éº„É´„ÉâÔºâ

    Args:
        pdb1, pdb2 (str): PDB„Éï„Ç°„Ç§„É´„Éë„Çπ

    Returns:
        float: RMSDÂÄ§ [√Ö]
    """
    parser = PDBParser(QUIET=True)
    structure1 = parser.get_structure('s1', pdb1)
    structure2 = parser.get_structure('s2', pdb2)

    # CŒ±ÂéüÂ≠ê„ÅÆ„Åø„ÇíÊäΩÂá∫
    atoms1 = []
    atoms2 = []

    for model in structure1:
        for chain in model:
            for residue in chain:
                if 'CA' in residue:
                    atoms1.append(residue['CA'])

    for model in structure2:
        for chain in model:
            for residue in chain:
                if 'CA' in residue:
                    atoms2.append(residue['CA'])

    # ÂéüÂ≠êÊï∞„ÇíÂêà„Çè„Åõ„ÇãÔºàÁü≠„ÅÑÊñπ„Å´ÊèÉ„Åà„ÇãÔºâ
    min_length = min(len(atoms1), len(atoms2))
    atoms1 = atoms1[:min_length]
    atoms2 = atoms2[:min_length]

    # ÊßãÈÄ†„ÇíÈáç„Å≠Âêà„Çè„Åõ
    super_imposer = Superimposer()
    super_imposer.set_atoms(atoms1, atoms2)

    rmsd = super_imposer.rms

    print(f"ÊßãÈÄ†ÊØîËºÉ:")
    print(f"  PDB1: {pdb1}")
    print(f"  PDB2: {pdb2}")
    print(f"  RMSD: {rmsd:.2f} √Ö")

    if rmsd < 2.0:
        print("  ‚Üí ÈùûÂ∏∏„Å´È°û‰ººÔºà„Åª„ÅºÂêå‰∏ÄÊßãÈÄ†Ôºâ")
    elif rmsd < 5.0:
        print("  ‚Üí È°û‰ººÔºàÂêå„Åò„Éï„Ç©„Éº„É´„ÉâÔºâ")
    else:
        print("  ‚Üí Áï∞„Å™„ÇãÊßãÈÄ†")

    return rmsd

import requests
import numpy as np
from Bio.PDB import PDBParser
from io import StringIO

class AlphaFoldAnalyzer:
    """AlphaFoldÊßãÈÄ†„ÅÆÂåÖÊã¨ÁöÑËß£Êûê„ÇØ„É©„Çπ"""

    def __init__(self, uniprot_id):
        """
        Args:
            uniprot_id (str): UniProt ID
        """
        self.uniprot_id = uniprot_id
        self.structure = None
        self.plddt_scores = None

    def download_structure(self):
        """ÊßãÈÄ†„Çí„ÉÄ„Ç¶„É≥„É≠„Éº„Éâ"""
        url = f"https://alphafold.ebi.ac.uk/files/AF-{self.uniprot_id}-F1-model_v4.pdb"

        try:
            response = requests.get(url, timeout=10)
            response.raise_for_status()

            parser = PDBParser(QUIET=True)
            self.structure = parser.get_structure(
                self.uniprot_id,
                StringIO(response.text)
            )

            print(f"‚úì ÊßãÈÄ†„ÉÄ„Ç¶„É≥„É≠„Éº„ÉâÊàêÂäü: {self.uniprot_id}")
            return True

        except Exception as e:
            print(f"‚úó „ÉÄ„Ç¶„É≥„É≠„Éº„ÉâÂ§±Êïó: {e}")
            return False

    def extract_plddt(self):
        """pLDDT„Çπ„Ç≥„Ç¢„ÇíÊäΩÂá∫"""
        if self.structure is None:
            print("‚úó ÊßãÈÄ†„ÅåÊú™„ÉÄ„Ç¶„É≥„É≠„Éº„Éâ")
            return None

        plddt = []
        for model in self.structure:
            for chain in model:
                for residue in chain:
                    for atom in residue:
                        if atom.name == 'CA':
                            plddt.append(atom.bfactor)
                            break

        self.plddt_scores = np.array(plddt)
        return self.plddt_scores

    def assess_quality(self):
        """‰∫àÊ∏¨ÂìÅË≥™„ÇíË©ï‰æ°"""
        if self.plddt_scores is None:
            self.extract_plddt()

        mean_plddt = np.mean(self.plddt_scores)
        very_high = np.sum(self.plddt_scores > 90) / len(self.plddt_scores) * 100
        confident = np.sum((self.plddt_scores >= 70) & (self.plddt_scores <= 90)) / len(self.plddt_scores) * 100
        low = np.sum(self.plddt_scores < 70) / len(self.plddt_scores) * 100

        quality_report = {
            'mean_plddt': mean_plddt,
            'very_high_pct': very_high,
            'confident_pct': confident,
            'low_pct': low,
            'overall_quality': self._get_quality_label(mean_plddt)
        }

        print("\nÂìÅË≥™Ë©ï‰æ°„É¨„Éù„Éº„Éà:")
        print(f"  Âπ≥ÂùápLDDT: {mean_plddt:.2f}")
        print(f"  Very high (>90): {very_high:.1f}%")
        print(f"  Confident (70-90): {confident:.1f}%")
        print(f"  Low (<70): {low:.1f}%")
        print(f"  Á∑èÂêàË©ï‰æ°: {quality_report['overall_quality']}")

        return quality_report

    def _get_quality_label(self, mean_plddt):
        """Á∑èÂêàÂìÅË≥™„É©„Éô„É´"""
        if mean_plddt > 90:
            return "ExcellentÔºàÂÆüÈ®ì„É¨„Éô„É´Ôºâ"
        elif mean_plddt > 80:
            return "Very goodÔºà„É¢„Éá„É™„É≥„Ç∞ÂèØËÉΩÔºâ"
        elif mean_plddt > 70:
            return "GoodÔºàÊ≥®ÊÑè„Åó„Å¶‰ΩøÁî®Ôºâ"
        else:
            return "PoorÔºà‰ø°È†ºÊÄß‰Ωé„ÅÑÔºâ"

    def find_flexible_regions(self, threshold=70):
        """ÊüîËªüÊÄß„ÅåÈ´ò„ÅÑÈ†òÂüüÔºà‰ΩépLDDTÔºâ„ÇíÊ§úÂá∫"""
        if self.plddt_scores is None:
            self.extract_plddt()

        flexible_regions = []
        in_region = False
        start = None

        for i, score in enumerate(self.plddt_scores):
            if score < threshold and not in_region:
                start = i + 1  # 1-indexed
                in_region = True
            elif score >= threshold and in_region:
                flexible_regions.append((start, i))
                in_region = False

        if in_region:
            flexible_regions.append((start, len(self.plddt_scores)))

        print(f"\nÊüîËªüÊÄßÈ†òÂüüÔºàpLDDT < {threshold}Ôºâ:")
        if flexible_regions:
            for start, end in flexible_regions:
                length = end - start + 1
                print(f"  ÊÆãÂü∫ {start}-{end} Ôºà{length}ÊÆãÂü∫Ôºâ")
        else:
            print("  „Å™„ÅóÔºàÂÖ®‰ΩìÁöÑ„Å´ÂâõÊÄß„ÅåÈ´ò„ÅÑÔºâ")

        return flexible_regions

    def get_summary(self):
        """ÂåÖÊã¨ÁöÑ„Çµ„Éû„É™„Éº"""
        if self.structure is None:
            self.download_structure()

        # Âü∫Êú¨ÊÉÖÂ†±
        num_residues = len(list(self.structure.get_residues()))

        # ÂìÅË≥™Ë©ï‰æ°
        quality = self.assess_quality()

        # ÊüîËªüÊÄßÈ†òÂüü
        flexible = self.find_flexible_regions()

        summary = {
            'uniprot_id': self.uniprot_id,
            'num_residues': num_residues,
            'quality': quality,
            'flexible_regions': flexible
        }

        return summary
