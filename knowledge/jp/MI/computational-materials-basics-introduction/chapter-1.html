def radial_wavefunction(r, n, l):
    """
    Ê∞¥Á¥†ÂéüÂ≠ê„ÅÆÂãïÂæÑÊ≥¢ÂãïÈñ¢Êï∞ R_nl(r)

    Args:
        r: ÂãïÂæÑÂ∫ßÊ®ô [√Ö]
        n: ‰∏ªÈáèÂ≠êÊï∞ (1, 2, 3, ...)
        l: Êñπ‰ΩçÈáèÂ≠êÊï∞ (0, 1, ..., n-1)

    Returns:
        R_nl(r)
    """
    rho = 2 * r / (n * a0)
    L = genlaguerre(n - l - 1, 2*l + 1)  # ‰∏ÄËà¨Âåñ„É©„Ç≤„Éº„É´Â§öÈ†ÖÂºè

    # Ê≠£Ë¶èÂåñÂÆöÊï∞
    N = np.sqrt((2/(n*a0))**3 * np.math.factorial(n-l-1) /
                (2*n*np.math.factorial(n+l)))

    R_nl = N * rho**l * np.exp(-rho/2) * L(rho)
    return R_nl

from scipy.linalg import eigh

def kronig_penney_band(N=100, a=1.0, b=0.2, V0=5.0, n_bands=5):
    """
    „ÇØ„É≠„Éº„Éã„ÉÉ„Éí„Éª„Éö„Éã„Éº„É¢„Éá„É´„ÅÆ„Éê„É≥„ÉâÊßãÈÄ†„ÇíË®àÁÆó

    Args:
        N: Âπ≥Èù¢Ê≥¢Âü∫Â∫ï„ÅÆÊï∞
        a: „Éù„ÉÜ„É≥„Ç∑„É£„É´„ÅÆÂπÖÔºà‰∫ïÊà∏Ôºâ [√Ö]
        b: „Éù„ÉÜ„É≥„Ç∑„É£„É´„ÅÆÂπÖÔºàÈöúÂ£ÅÔºâ [√Ö]
        V0: ÈöúÂ£Å„ÅÆÈ´ò„Åï [eV]
        n_bands: Ë°®Á§∫„Åô„Çã„Éê„É≥„ÉâÊï∞

    Returns:
        k_points: Ê≥¢Êï∞„Éô„ÇØ„Éà„É´
        energies: „Ç®„Éç„É´„ÇÆ„Éº„Éê„É≥„Éâ
    """
    d = a + b  # Âë®Êúü
    G = 2 * np.pi / d  # ÈÄÜÊ†ºÂ≠ê„Éô„ÇØ„Éà„É´

    # Á¨¨‰∏Ä„Éñ„É™„É´„Ç¢„É≥„Çæ„Éº„É≥ÂÜÖ„ÅÆkÁÇπ
    k_points = np.linspace(-np.pi/d, np.pi/d, 200)
    energies = np.zeros((len(k_points), n_bands))

    # „Éù„ÉÜ„É≥„Ç∑„É£„É´„ÅÆ„Éï„Éº„É™„Ç®‰øÇÊï∞ÔºàÁü©ÂΩ¢„Éù„ÉÜ„É≥„Ç∑„É£„É´Ôºâ
    def V_G(n):
        if n == 0:
            return V0 * b / d
        else:
            return V0 * np.sin(n * G * b / 2) / (n * np.pi) * np.exp(-1j * n * G * (a + b/2))

    for ik, k in enumerate(k_points):
        # „Éè„Éü„É´„Éà„Éã„Ç¢„É≥Ë°åÂàó„ÅÆÊßãÁØâÔºàÂπ≥Èù¢Ê≥¢Âü∫Â∫ïÔºâ
        H = np.zeros((2*N+1, 2*N+1), dtype=complex)

        for i in range(-N, N+1):
            for j in range(-N, N+1):
                if i == j:
                    # ÂØæËßíÊàêÂàÜ: ÈÅãÂãï„Ç®„Éç„É´„ÇÆ„Éº
                    H[i+N, j+N] = 0.5 * (k + i*G)**2  # ÂéüÂ≠êÂçò‰ΩçÁ≥ª
                else:
                    # ÈùûÂØæËßíÊàêÂàÜ: „Éù„ÉÜ„É≥„Ç∑„É£„É´
                    H[i+N, j+N] = V_G(i - j)

        # Âõ∫ÊúâÂÄ§ÂïèÈ°å„ÇíËß£„Åè
        eigvals, eigvecs = eigh(H)
        energies[ik, :] = eigvals[:n_bands]

    return k_points, energies

import numpy as np
import matplotlib.pyplot as plt

def quantum_well(L=10.0, n_states=5):
    """
    ÁÑ°Èôê‰∫ïÊà∏Âûã„Éù„ÉÜ„É≥„Ç∑„É£„É´„ÅÆÊ≥¢ÂãïÈñ¢Êï∞„Å®„Ç®„Éç„É´„ÇÆ„ÉºÊ∫ñ‰Ωç

    Args:
        L: ‰∫ïÊà∏„ÅÆÂπÖ [√Ö]
        n_states: Ë°®Á§∫„Åô„ÇãÁä∂ÊÖãÊï∞

    Returns:
        x: ‰ΩçÁΩÆÂ∫ßÊ®ô
        psi: Ê≥¢ÂãïÈñ¢Êï∞
        E: „Ç®„Éç„É´„ÇÆ„ÉºÊ∫ñ‰Ωç
    """
    x = np.linspace(0, L, 500)

    # „Ç®„Éç„É´„ÇÆ„ÉºÊ∫ñ‰ΩçÔºàeVÂçò‰Ωç„Å´Â§âÊèõÔºâ
    # hbar^2 / (2*m_e) = 3.81 eV √Ö^2ÔºàÂéüÂ≠êÂçò‰ΩçÁ≥ªÔºâ
    coeff = 3.81
    E = np.array([coeff * (n*np.pi/L)**2 for n in range(1, n_states+1)])

    # Ê≥¢ÂãïÈñ¢Êï∞
    psi = np.zeros((n_states, len(x)))
    for n in range(1, n_states+1):
        psi[n-1, :] = np.sqrt(2/L) * np.sin(n * np.pi * x / L)

    return x, psi, E
