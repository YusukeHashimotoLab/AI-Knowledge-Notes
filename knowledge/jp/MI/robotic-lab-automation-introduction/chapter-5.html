    """

    def synthesize_qd(self, color, cd_se_ratio, temperature, time):
        """
        é‡å­ãƒ‰ãƒƒãƒˆåˆæˆ

            'cd_se_ratio': cd_se_ratio,
            'temperature': temperature,
            'time': time,
            'emission': emission,
            'qy': qy
        })

        return emission, qy

    def optimize_rgb(self, iterations=20):
        """
        RGB 3è‰²ã‚’åŒæ™‚æœ€é©åŒ–

            }
